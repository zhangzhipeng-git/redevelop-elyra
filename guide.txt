

## 前端
# 进入项目目录
cd D:\jupyterlab\redevelop_elyra

# 激活环境
conda activate dev

# 构建项目
jlpm run build

# 建立软链接，不需要重新安装插件 pip install -e .
jupyter labextension develop --overwrite .

# 构建/热更新
jlpm run build / jlpm run watch

# 打开编辑器
jupyter lab


## 整个编辑器汉化
# 安装简体中文语言包
pip install jupyterlab-language-pack-zh-CN
# 去编辑器的settings里设置language

## 打包部署
# 打包 & 安装
pip install build
python -m build
pip install jupyterlab_apod/dist/jupyterlab_apod-0.1.0-py3-none-any.whl
# 发布 npm 包 & 下载编译安装
